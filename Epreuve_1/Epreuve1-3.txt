G : tourner à gauche
D : tourner à droite
T : aller tout droit (1case)

chaîne d’ordres : GTDTDT
 
SortirLabyrinthe()
tourner à gauche "90°"
chaîne d’ordres <- ajout G 

si (T = false)
	tourne à droite "90°"
	chaîne d’ordres <- chaîne d’ordres + D

	si (T = false)
		tourner à droite "90°"
		chaîne d’ordres <- chaîne d’ordres + D
		
	sinon
		avanvce 1 case
		chaîne d’ordres <- chaîne d’ordres + T
		return SortirLabyrinthe

si (T = true)
	avance 1 case
	chaîne d’ordres <- chaîne d’ordres + T
	return SortirLabyrinthe
